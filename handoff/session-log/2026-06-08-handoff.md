```typescript
class AgentRegistry {
  register(agentId: string, agent: AgentContext): void
  unregister(agentId: string): void
  get(agentId: string): AgentContext | undefined
  has(agentId: string): boolean
  list(): string[]
}
```

### AgentRouter

```typescript
class AgentRouter {
  constructor(registry: AgentRegistry, tracer?: ProteusTracer, costTracker?: CostAttributionTracker)

}
```

```typescript
class SubHarness extends Harness {

}
```
