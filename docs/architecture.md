```
│  │     └─ 深度只读快照 + 时间戳 + SHA-256 校验和                       │
│  │                                                                      │
│  ├─► CheckpointLog.saveCheckpoint(frozenContext)                        │
│  │                                                                      │
│  ├─► [O] OTel: turn span end + turn.duration 指标                      │
│  │                                                                      │
│  └─► return TurnResult { status, turnId }                              │
│                                                                         │
└─────────────────────────────────────────────────────────────────────────┘
    │
    ▼
┌─────────────────────────────────────────────────────────────────────────┐
│                     Harness.runChain() 循环控制                          │
│                                                                         │
│  ┌─────────────────────────────────────────────────────────────┐       │
│  │  while (turns < maxTurns) {                                 │       │
│  │    result = runTurn(session, agent)                         │       │
│  │                                                             │       │
│  │    switch (result.status) {                                 │       │
│  │      case "completed":                                      │       │
│  │        if (no more toolCalls) → return "completed"          │       │
│  │        else → continue loop (下一轮 Turn)                   │       │
│  │                                                             │       │
│  │      case "suspended":                                      │       │
│  │        → return "suspended" (外部输入到来后 resume)         │       │
│  │                                                             │       │
│  │      case "aborted":                                        │       │
│  │        → return "aborted" (用户中止)                        │       │
│  │                                                             │       │
│  │      case "errored":                                        │       │
│  │        → return "errored" (可恢复 → 下轮重试)               │       │
│  │    }                                                        │       │
│  │  }                                                          │       │
│  │  return "max_turns"                                         │       │
│  └─────────────────────────────────────────────────────────────┘       │
│                                                                         │
│  chain:end 事件 ────────────────────────────────────────────────► OTel  │
│                                                                         │
└─────────────────────────────────────────────────────────────────────────┘
```

## 五、Checkpoint / Resume 流程

```
┌─────────────────────────────────────────────────────────────────────────┐
│                         Checkpoint 保存                                  │
│                                                                         │
│  turn:end 事件触发                                                      │
│  │                                                                      │
│  ├─► HandlerContext.freeze()                                           │
│  │     ├─ 深拷贝 TurnContext (messages, toolResults, metadata)         │
│  │     ├─ 快照 SessionContext (messages, cost, config)                 │
│  │     ├─ 生成 timestamp                                               │
│  │     └─ 计算 SHA-256 checksum                                        │
│  │                                                                      │
│  └─► CheckpointLog.saveCheckpoint(frozenContext)                        │
│        └─ SQLite: INSERT INTO checkpoints (session_id, turn_id, ...)   │
│                                                                         │
│  suspend 时额外保存:                                                    │
│  └─ frozenContext + { resumeReason: "suspend", pendingInput: ... }     │
│                                                                         │
└─────────────────────────────────────────────────────────────────────────┘

┌─────────────────────────────────────────────────────────────────────────┐
│                         Resume 恢复                                     │
│                                                                         │
│  Harness.resume(sessionId, agent, input?)                              │
│  │                                                                      │
│  ├─ ① CheckpointLog.loadLatestCheckpoint(sessionId)                    │
│  │     └─ → FrozenContext                                              │
│  │                                                                      │
│  ├─ ② 从 FrozenContext 重建 SessionContext                             │
│  │     ├─ 恢复 WorkingMemory (messages)                                │
│  │     ├─ 恢复 CostTracker (cost)                                      │
│  │     └─ 恢复 SessionConfig                                           │
│  │                                                                      │
│  ├─ ③ 从 FrozenContext 重建 TurnContext                                │
│  │     ├─ 恢复 messages, toolResults, metadata                         │
│  │     └─ 定位到 checkpoint.phaseIndex + 1                             │
│  │                                                                      │
│  ├─ ④ 如果 resumeReason === "suspend":                                │
│  │     └─ 将 input 写入 TurnContext (pendingInput 解析)                │
│  │                                                                      │
│  └─ ⑤ 从下一个阶段继续执行                                             │
│        └─ emit("phase:before", { phaseName: PHASES[phaseIndex+1] })   │
│                                                                         │
└─────────────────────────────────────────────────────────────────────────┘
```
